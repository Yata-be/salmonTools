import matplotlib.pyplot as plt def click(*ags, **kws): waves = int(Element("waves").element.value) weapons = int(Element("weapons").element.value) Coef = [0.2 + i * 0.016 for i in range(51)] # 確率係数 Prob = [[0]*51] Prob[0][0] = 1 for i in range(700): newProb = [] for j in range(51): if j == 0: newProb.append(Prob[-1][j] * Coef[j]) else: newProb.append(Prob[-1][j] * Coef[j] + Prob[-1][j-1] * (1 - Coef[j-1])) Prob.append(newProb) y = [] for i in range(701): s = sum(Prob[i][weapons:]) y.append(s) if s > 1 - 10**-5: break x = [i for i in range(len(y))] pyscript.write("print", str(waves) + "Wavesで" + str(weapons) + "種類以上コンプできる確率: "+ str(y[waves])) fig, ax = plt.subplots() ax.plot(x, y) ax.hlines(y[waves], 0, len(y)) pyscript.write("plot", fig)
